export class AuthService {
  constructor(
    private usersRepo: UsersRepository,

  ) {}

    const hash = await bcrypt.hash(dto.password, 10);

    const user = await this.usersRepo.create({
      email: dto.email,
      phone: dto.phone,
      passwordHash: hash,
      role: dto.role,
      isVerified: false,
    });

    await this.sendOtp(dto.phone);

    return { message: 'OTP sent to phone', userId: user.id };
  }

  async verifyOtp(userId: string, code: string) {
    const valid = await this.validateOtp(userId, code);

    if (!valid) throw new Error('Invalid OTP');

    if (!match) throw new Error('Invalid credentials');

    const token = this.jwtService.sign({ id: user.id, role: user.role });

@Injectable()
export class CampaignService {
  constructor(
    private campaignRepo: CampaignRepository,
    private pricingService: PricingService,
  ) {}

  async createDraft(userId: string, dto: CreateCampaignDto) {
    return this.campaignRepo.create({
      userId,
      type: dto.type,
      status: 'DRAFT',
      textContent: dto.textContent,
      mediaUrls: dto.mediaUrls || [],
      videoUrl: dto.videoUrl || null,
    });
  }

  async calculatePricing(campaignId: string) {
    const campaign = await this.campaignRepo.findById(campaignId);

    const price = await this.pricingService.calculate({
      type: campaign.type,
      durationDays: this.getDuration(campaign),
      frequency: campaign.frequency,
    });

    await this.campaignRepo.update(campaignId, {
      totalCost: price.finalPrice,
    });

    return price;
  }

  async submitCampaign(campaignId: string) {
    return this.campaignRepo.update(campaignId, {
      status: 'PENDING_PAYMENT',
    });
  }

  private getDuration(campaign: Campaign) {
    return (
      (campaign.endDate.getTime() - campaign.startDate.getTime()) /
      (1000 * 60 * 60 * 24)
    );
  }
}

## 3.1 Pricing Service

```ts
@Injectable()
export class PricingService {
  async calculate(input: {
    type: 'TEXT' | 'IMAGE' | 'VIDEO';
    durationDays: number;
    frequency: string;
  }) {
    let base = this.getBasePrice(input.type);

    let durationMultiplier = this.getDurationMultiplier(input.durationDays);

    let frequencyMultiplier = this.getFrequencyMultiplier(input.frequency);

    const finalPrice = base * durationMultiplier * frequencyMultiplier;

    return {
      base,
      durationMultiplier,
      frequencyMultiplier,
      finalPrice,
    };
  }

  private getBasePrice(type: string) {
    switch (type) {
      case 'TEXT':
        return 399;
      case 'IMAGE':
        return 599;
      case 'VIDEO':
        return 899;
    }
  }

  private getDurationMultiplier(days: number) {
    if (days <= 7) return 1;
    if (days <= 14) return 1.3;
    return 1.6;
  }

  private getFrequencyMultiplier(freq: string) {
    switch (freq) {
      case 'ALL_DAY':
        return 1.8;
      case 'TWICE_DAILY':
        return 1.4;
      case 'THRICE_WEEK':
        return 1.1;
      case 'ONCE_WEEK':
        return 1;
    }
  }
}
```

@Injectable()
export class PaymentService {
  constructor(private paymentRepo: PaymentRepository) {}

  async initiatePayment(dto: InitiatePaymentDto) {
    const payment = await this.paymentRepo.create({
      campaignId: dto.campaignId,
      method: dto.method,
      amount: dto.amount,
      status: 'PENDING',
    });

    if (dto.method === 'MPESA') {
      return this.initiateMpesa(dto.phone, dto.amount, payment.id);
    }

    if (dto.method === 'PAYPAL') {
      return this.initiatePaypal(dto.amount, payment.id);
    }

    return payment;
  }

  async confirmPayment(transactionRef: string) {
    return this.paymentRepo.updateByRef(transactionRef, {
      status: 'SUCCESS',
    });
  }

  private async initiateMpesa(phone: string, amount: number, ref: string) {
    // STK push integration
    return { message: 'STK push sent', ref };
  }

  private async initiatePaypal(amount: number, ref: string) {
    // PayPal API integration
    return { redirectUrl: 'paypal-checkout-url', ref };
  }
}
